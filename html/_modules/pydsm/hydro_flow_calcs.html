[docs]def read_ts(row,base_dir):
    '''
    Each row should have :- 
    * a 'NAME' column
    * a 'SIGN' column
    * a 'FILE' column
    * a 'PATH' column
    '''
    if (row['FILE'].lower()=='constant'):
        return float(row['PATH'])
    df,units,ptype = next(pyhecdss.get_ts(os.path.join(base_dir,row['FILE']),row['PATH']))
    df.columns=[row['NAME']]
    factor = float(row['SIGN'])
    df.iloc[:,0]=factor*df.iloc[:,0]
    return df



[docs]def sum_paths(df,base_dir):
    sum=0.0
    for idx,row in df.iterrows():
        df=read_ts(row,base_dir)
        sum+=df.iloc[:,0]
    return sum

[docs]def get_outflow_chipps(tidefile):
    hydro=pydsm.hydroh5.HydroH5(os.path.join(dsm2_output_dir,tidefile))
    flow437=hydro.get_channel_flow('437','upstream')
    flow551=hydro.get_channel_flow('511','downstream')
    flow = flow437.iloc[:,0]+flow551.iloc[:,0]
    flow = flow.to_frame()
    flow.columns=['out_chipps']
    return flow
